import React, { useCallback, useEffect } from 'react';
import {
  ReactFlow,
  ReactFlowProvider,
  Background,
  BackgroundVariant,
  Controls,
  Connection,
  Edge,
  Node,
  addEdge,
  useNodesState,
  useEdgesState,
  NodeTypes,
} from '@xyflow/react';
import '@xyflow/react/dist/style.css';
import { useSelector } from 'react-redux';
import { RootState } from '../../store/store';
 
type TaskStatus = 'pending' | 'started' | 'completed' | 'error';
type TaskType = 'PROJECT::STATE:SETUP' | 'PM:PRD::ANALYSIS' | 'PM:UXSMD::ANALYSIS' | 'WEBAPP:COMPONENT::GENERATE';
 
interface TaskNodeData {
  label: string;
  status: TaskStatus;
  message?: string;
  [key: string]: unknown;
}
 
type CustomNode = Node<TaskNodeData>;
type CustomEdge = Edge;
 
const nodeTypes: Record<TaskType, { style: { background: string } }> = {
  'PROJECT::STATE:SETUP': { style: { background: '#e3f2fd' } },
  'PM:PRD::ANALYSIS': { style: { background: '#f3e5f5' } },
  'PM:UXSMD::ANALYSIS': { style: { background: '#e8f5e9' } },
  'WEBAPP:COMPONENT::GENERATE': { style: { background: '#fff3e0' } },
};
 
const Flow: React.FC = () => {
  const { messages, taskUpdates } = useSelector((state: RootState) => state.ai);
  const [nodes, setNodes, onNodesChange] = useNodesState<CustomNode>([]);
  const [edges, setEdges, onEdgesChange] = useEdgesState<CustomEdge>([]);
 
  useEffect(() => {
    if (!taskUpdates) return;
 
    const newNodes: CustomNode[] = Object.values(taskUpdates).map((task, index) => ({
      id: task.type,
      type: 'default',
      data: {
        label: task.type.split('::').pop() || task.type,
        status: task.status as TaskStatus,
        message: task.message,
      },
      position: { x: index * 200, y: 100 },
      style: {
        ...(nodeTypes[task.type as TaskType]?.style || { background: '#f5f5f5' }),
        border: '1px solid #ccc',
        padding: 10,
        borderRadius: 5,
        width: 180,
        ...(task.status === 'started' && { borderColor: '#2196f3', borderWidth: 2 }),
        ...(task.status === 'completed' && { borderColor: '#4caf50', borderWidth: 2 }),
        ...(task.status === 'error' && { borderColor: '#f44336', borderWidth: 2 }),
      },
    }));
 
    setNodes(newNodes);
 
    // Create edges between nodes based on parent-child relationships
    const newEdges: CustomEdge[] = [];
    Object.values(taskUpdates).forEach((task) => {
      if (task.children) {
        task.children.forEach((childType) => {
          newEdges.push({
            id: `${task.type}-${childType}`,
            source: task.type,
            target: childType,
            type: 'smoothstep',
          });
        });
      }
    });
 
    setEdges(newEdges);
  }, [taskUpdates, setNodes, setEdges]);
 
  const onConnect = useCallback(
    (params: Connection) => setEdges((eds) => addEdge(params, eds as CustomEdge[])),
    [setEdges]
  );
 
  return (
    <div className="h-[600px] w-full bg-white rounded-lg shadow-md">
      <ReactFlow
        nodes={nodes}
        edges={edges}
        onNodesChange={onNodesChange}
        onEdgesChange={onEdgesChange}
        onConnect={onConnect}
        fitView
      >
        <Background variant={BackgroundVariant.Dots} />
        <Controls />
      </ReactFlow>
    </div>
  );
};
 
// Wrap the Flow component with ReactFlowProvider
const AIFlowVisualization: React.FC = () => {
  return (
    <ReactFlowProvider>
      <Flow />
    </ReactFlowProvider>
  );
};
 
export default AIFlowVisualization;
